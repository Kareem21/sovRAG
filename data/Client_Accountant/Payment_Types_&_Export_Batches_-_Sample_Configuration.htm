Payment Types & Export Batches - Sample Configuration

The following sample was set up based on the requirements of a specific HSBC branch.

 

 

[/PAYSTART=<?xml version="1.0" encoding="utf-8"?>

<Document>

  <pain.001.001.02>

    <GrpHdr>

      <MsgId>Payment Batch #(+PEB.BatchNr+)</MsgId>

      <CreDtTm>(+PEB.DateExport\YYYY-MM-DD\Thh:mm:ss+)</CreDtTm>

      <NbOfTxs>(+PEB.TotalTrans+)</NbOfTxs>

      <Grpg>MIXD</Grpg>

      <InitgPty>

        <Id>

          <OrgId>

            <BkPtyId>(+CLIENT_BANK_PAYMENT_ID+)</BkPtyId>

          </OrgId>

        </Id>

      </InitgPty>

    </GrpHdr>

/]

{PAYEND Sample 1}:

[/PAYEND=</pain.001.001.02>

</Document>

/]

{PAYEND Sample 2}:

[/PAYEND=<control>

  <recordType>9</recordType>

  <numTrans>(+PEB.TotalTrans+)</numTrans>

  <value>(+PSB.TotalAmount+)</value>

</control>

</importfile/>

/]

[/PAYINFO=<PmtInf>

  <PmtInfId>VP Payment #(+PAY.PaymentId+)</PmtInfId>

  <PmtMtd>TRF</PmtMtd>

  <PmtTpInf>

<ClrChanl>MPNS</ClrChanl>

  </PmtTpInf>

  <ReqdExctnDt>(+PAY.PayDate\YYYY-MM-DD+)</ReqdExctnDt>

  <Dbtr>

<Nm>(+PAY.EntCode\#AD-1-0+)</Nm>

  </Dbtr>

  <DbtrAcct>

<Id>

  <PrtryAcct>

<Id>(+PSB.ExtAcctNo+)</Id>

  </PrtryAcct>

</Id>

  </DbtrAcct>

  <DbtrAgt>

<FinInstnId>

  <CmbndId>

<ClrSysMmbId>

  <Id>(+HSBC_BANK_ID+)</Id>

</ClrSysMmbId>

<PstlAdr>

  <Ctry>(+HSBC_BANK_CNTR+)</Ctry>

</PstlAdr>

  </CmbndId>

</FinInstnId>

  </DbtrAgt>

  <CdtTrfTxInf>

<PmtId>

  <EndToEndId>(+PAY.PayReference+)</EndToEndId>

</PmtId>

<Amt>

  <InstdAmt Ccy="(+PAY.PayCCY+)">(+PAY.PayAmount+)</InstdAmt>

</Amt>

<CdtrAgt>

  <FinInstnId>

<CmbndId>

  <ClrSysMmbId>

<Id>(+PAY.BankIdCode+)</Id>

  </ClrSysMmbId>

  <PstlAdr>

<Ctry>(+PAY.BankCNTR+)</Ctry>

  </PstlAdr>

</CmbndId>

  </FinInstnId>

</CdtrAgt>

<Cdtr>

  <Nm>(+PAY.PayeeName+)</Nm>

</Cdtr>

<CdtrAcct>

  <Id>

<PrtryAcct>

  <Id>(+PAY.BankAccount+)</Id>

</PrtryAcct>

  </Id>

</CdtrAcct>

  </CdtTrfTxInf>

</PmtInf>

/]
